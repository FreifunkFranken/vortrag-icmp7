            \begin{itemize}
                \item<1-> git clone git://git.freifunk-ol.de/ffol/user/reddog/firmware.git
                \item<1-> cd firmware
                \item<1-> ./buildscript
                \item<3-> ./buildscript selectbsp bsp/board\_wr1043nd.bsp
